333  template<typename T,
334  typename = std::enable_if_t<std::is_same<T, double>::value ||
335  std::is_same<T, std::complex<double>>::value>>
336  void encode_internal(const std::vector<T> &values,
337  parms_id_type parms_id, double scale, Plaintext &destination,
338  MemoryPoolHandle pool)
339  {
340  // Verify parameters.
341  auto context_data_ptr = context_->context_data(parms_id);
342  if (!context_data_ptr)
343  {
344  throw std::invalid_argument("parms_id is not valid for encryption parameters");
345  }
346  if (values.size() > slots_)
347  {
348  throw std::invalid_argument("values has invalid size");
349  }
350  if (!pool)
351  {
352  throw std::invalid_argument("pool is uninitialized");
353  }
354 
355  auto &context_data = *context_data_ptr;
356  auto &parms = context_data.parms();
357  auto &coeff_modulus = parms.coeff_modulus();
358  std::size_t coeff_mod_count = coeff_modulus.size();
359  std::size_t coeff_count = parms.poly_modulus_degree();
360 
361  // Quick sanity check
362  if (!util::product_fits_in(coeff_mod_count, coeff_count))
363  {
364  throw std::logic_error("invalid parameters");
365  }
366 
367  // Check that scale is positive and not too large
368  if (scale <= 0 || (static_cast<int>(log2(scale)) + 1 >=
369  context_data.total_coeff_modulus_bit_count()))
370  {
371  throw std::invalid_argument("scale out of bounds");
372  }
373 
374  auto &small_ntt_tables = context_data.small_ntt_tables();
375 
376  // input_size is guaranteed to be no bigger than slots_
377  std::size_t input_size = values.size();
378  std::size_t n = util::mul_safe(slots_, std::size_t(2));
379 
380  auto conj_values = util::allocate<std::complex<double>>(n, pool, 0);
381  for (std::size_t i = 0; i < input_size; i++)
382  {
383  conj_values[matrix_reps_index_map_[i]] = values[i];
384  conj_values[matrix_reps_index_map_[i + slots_]] = std::conj(values[i]);
385  }
386 
387  int logn = util::get_power_of_two(n);
388  std::size_t tt = 1;
389  for (int i = 0; i < logn; i++)
390  {
391  std::size_t mm = std::size_t(1) << (logn - i);
392  std::size_t k_start = 0;
393  std::size_t h = mm / 2;
394 
395  for (std::size_t j = 0; j < h; j++)
396  {
397  std::size_t k_end = k_start + tt;
398  auto s = inv_roots_[h + j];
399 
400  for (std::size_t k = k_start; k < k_end; k++)
401  {
402  auto u = conj_values[k];
403  auto v = conj_values[k + tt];
404  conj_values[k] = u + v;
405  conj_values[k + tt] = (u - v) * s;
406  }
407 
408  k_start += 2 * tt;
409  }
410  tt *= 2;
411  }
412 
413  double n_inv = double(1.0) / static_cast<double>(n);
414 
415  // Put the scale in at this point
416  n_inv *= scale;
417 
418  int max_coeff_bit_count = 1;
419  for (std::size_t i = 0; i < n; i++)
420  {
421  // Multiply by scale and n_inv (see above)
422  conj_values[i] *= n_inv;
423 
424  // Verify that the values are not too large to fit in coeff_modulus
425  // Note that we have an extra + 1 for the sign bit
426  max_coeff_bit_count = std::max(max_coeff_bit_count,
427  static_cast<int>(std::log2(std::fabs(conj_values[i].real()))) + 2);
428  }
429  if (max_coeff_bit_count >= context_data.total_coeff_modulus_bit_count())
430  {
431  throw std::invalid_argument("encoded values are too large");
432  }
433 
434  double two_pow_64 = std::pow(2.0, 64);
435 
436  // Resize destination to appropriate size
437  // Need to first set parms_id to zero, otherwise resize
438  // will throw an exception.
439  destination.parms_id() = parms_id_zero;
440  destination.resize(util::mul_safe(coeff_count, coeff_mod_count));
441 
442  // Use faster decomposition methods when possible
443  if (max_coeff_bit_count <= 64)
444  {
445  for (std::size_t i = 0; i < n; i++)
446  {
447  double coeffd = std::round(conj_values[i].real());
448  bool is_negative = std::signbit(coeffd);
449 
450  std::uint64_t coeffu =
451  static_cast<std::uint64_t>(std::fabs(coeffd));
452 
453  if (is_negative)
454  {
455  for (std::size_t j = 0; j < coeff_mod_count; j++)
456  {
457  destination[i + (j * coeff_count)] = util::negate_uint_mod(
458  coeffu % coeff_modulus[j].value(), coeff_modulus[j]);
459  }
460  }
461  else
462  {
463  for (std::size_t j = 0; j < coeff_mod_count; j++)
464  {
465  destination[i + (j * coeff_count)] =
466  coeffu % coeff_modulus[j].value();
467  }
468  }
469  }
470  }
471  else if (max_coeff_bit_count <= 128)
472  {
473  for (std::size_t i = 0; i < n; i++)
474  {
475  double coeffd = std::round(conj_values[i].real());
476  bool is_negative = std::signbit(coeffd);
477  coeffd = std::fabs(coeffd);
478 
479  std::uint64_t coeffu[2]{
480  static_cast<std::uint64_t>(std::fmod(coeffd, two_pow_64)),
481  static_cast<std::uint64_t>(coeffd / two_pow_64) };
482 
483  if (is_negative)
484  {
485  for (std::size_t j = 0; j < coeff_mod_count; j++)
486  {
487  destination[i + (j * coeff_count)] =
488  util::negate_uint_mod(util::barrett_reduce_128(
489  coeffu, coeff_modulus[j]), coeff_modulus[j]);
490  }
491  }
492  else
493  {
494  for (std::size_t j = 0; j < coeff_mod_count; j++)
495  {
496  destination[i + (j * coeff_count)] =
497  util::barrett_reduce_128(coeffu, coeff_modulus[j]);
498  }
499  }
500  }
501  }
502  else
503  {
504  // Slow case
505  auto coeffu(util::allocate_uint(coeff_mod_count, pool));
506  auto decomp_coeffu(util::allocate_uint(coeff_mod_count, pool));
507  for (std::size_t i = 0; i < n; i++)
508  {
509  double coeffd = std::round(conj_values[i].real());
510  bool is_negative = std::signbit(coeffd);
511  coeffd = std::fabs(coeffd);
512 
513  // We are at this point guaranteed to fit in the allocated space
514  util::set_zero_uint(coeff_mod_count, coeffu.get());
515  auto coeffu_ptr = coeffu.get();
516  while (coeffd >= 1)
517  {
518  *coeffu_ptr++ = static_cast<std::uint64_t>(
519  std::fmod(coeffd, two_pow_64));
520  coeffd /= two_pow_64;
521  }
522 
523  // Next decompose this coefficient
524  decompose_single_coeff(context_data, coeffu.get(),
525  decomp_coeffu.get(), pool);
526 
527  // Finally replace the sign if necessary
528  if (is_negative)
529  {
530  for (std::size_t j = 0; j < coeff_mod_count; j++)
531  {
532  destination[i + (j * coeff_count)] =
533  util::negate_uint_mod(decomp_coeffu[j], coeff_modulus[j]);
534  }
535  }
536  else
537  {
538  for (std::size_t j = 0; j < coeff_mod_count; j++)
539  {
540  destination[i + (j * coeff_count)] = decomp_coeffu[j];
541  }
542  }
543  }
544  }
545 
546  // Transform to NTT domain
547  for (std::size_t i = 0; i < coeff_mod_count; i++)
548  {
549  util::ntt_negacyclic_harvey(
550  destination.data(i * coeff_count), small_ntt_tables[i]);
551  }
552 
553  destination.parms_id() = parms_id;
554  destination.scale() = scale;
555  }

557  template<typename T,
558  typename = std::enable_if_t<std::is_same<T, double>::value ||
559  std::is_same<T, std::complex<double>>::value>>
560  void decode_internal(const Plaintext &plain, std::vector<T> &destination,
561  MemoryPoolHandle pool)
562  {
563  // Verify parameters.
564  if (!plain.is_valid_for(context_))
565  {
566  throw std::invalid_argument("plain is not valid for encryption parameters");
567  }
568  if (!plain.is_ntt_form())
569  {
570  throw std::invalid_argument("plain is not in NTT form");
571  }
572  if (!pool)
573  {
574  throw std::invalid_argument("pool is uninitialized");
575  }
576 
577  auto context_data_ptr = context_->context_data(plain.parms_id());
578  auto &parms = context_data_ptr->parms();
579  auto &coeff_modulus = parms.coeff_modulus();
580  std::size_t coeff_mod_count = coeff_modulus.size();
581  std::size_t coeff_count = parms.poly_modulus_degree();
582  std::size_t rns_poly_uint64_count =
583  util::mul_safe(coeff_count, coeff_mod_count);
584 
585  auto &small_ntt_tables = context_data_ptr->small_ntt_tables();
586 
587  // Check that scale is positive and not too large
588  if (plain.scale() <= 0 || (static_cast<int>(log2(plain.scale())) >=
589  context_data_ptr->total_coeff_modulus_bit_count()))
590  {
591  throw std::invalid_argument("scale out of bounds");
592  }
593 
594  auto decryption_modulus = context_data_ptr->total_coeff_modulus();
595  auto upper_half_threshold = context_data_ptr->upper_half_threshold();
596 
597  auto &inv_coeff_products_mod_coeff_array =
598  context_data_ptr->base_converter()->get_inv_coeff_mod_coeff_array();
599  auto coeff_products_array =
600  context_data_ptr->base_converter()->get_coeff_products_array();
601 
602  int logn = util::get_power_of_two(coeff_count);
603 
604  // Quick sanity check
605  if ((logn < 0) || (coeff_count < SEAL_POLY_MOD_DEGREE_MIN) ||
606  (coeff_count > SEAL_POLY_MOD_DEGREE_MAX))
607  {
608  throw std::logic_error("invalid parameters");
609  }
610 
611  double inv_scale = double(1.0) / plain.scale();
612 
613  // Create mutable copy of input
614  auto plain_copy = util::allocate_uint(rns_poly_uint64_count, pool);
615  util::set_uint_uint(plain.data(), rns_poly_uint64_count, plain_copy.get());
616 
617  // Array to keep number bigger than std::uint64_t
618  auto temp(util::allocate_uint(coeff_mod_count, pool));
619 
620  // destination mod q
621  auto wide_tmp_dest(util::allocate_zero_uint(rns_poly_uint64_count, pool));
622 
623  // Transform each polynomial from NTT domain
624  for (std::size_t i = 0; i < coeff_mod_count; i++)
625  {
626  util::inverse_ntt_negacyclic_harvey(
627  plain_copy.get() + (i * coeff_count), small_ntt_tables[i]);
628  }
629 
630  auto res = util::allocate<std::complex<double>>(coeff_count, pool);
631 
632  double two_pow_64 = std::pow(2.0, 64);
633  for (std::size_t i = 0; i < coeff_count; i++)
634  {
635  for (std::size_t j = 0; j < coeff_mod_count; j++)
636  {
637  std::uint64_t tmp = util::multiply_uint_uint_mod(
638  plain_copy[(j * coeff_count) + i],
639  inv_coeff_products_mod_coeff_array[j], // (qi/q * plain[i]) mod qi
640  coeff_modulus[j]);
641  util::multiply_uint_uint64(
642  coeff_products_array + (j * coeff_mod_count),
643  coeff_mod_count, tmp, coeff_mod_count, temp.get());
644  util::add_uint_uint_mod(temp.get(),
645  wide_tmp_dest.get() + (i * coeff_mod_count),
646  decryption_modulus, coeff_mod_count,
647  wide_tmp_dest.get() + (i * coeff_mod_count));
648  }
649 
650  res[i] = 0.0;
651  if (util::is_greater_than_or_equal_uint_uint(
652  wide_tmp_dest.get() + (i * coeff_mod_count),
653  upper_half_threshold, coeff_mod_count))
654  {
655  double scaled_two_pow_64 = inv_scale;
656  for (std::size_t j = 0; j < coeff_mod_count;
657  j++, scaled_two_pow_64 *= two_pow_64)
658  {
659  if (wide_tmp_dest[i * coeff_mod_count + j] > decryption_modulus[j])
660  {
661  auto diff = wide_tmp_dest[i * coeff_mod_count + j] - decryption_modulus[j];
662  res[i] += diff ?
663  static_cast<double>(diff) * scaled_two_pow_64 : 0.0;
664  }
665  else
666  {
667  auto diff = decryption_modulus[j] - wide_tmp_dest[i * coeff_mod_count + j];
668  res[i] -= diff ?
669  static_cast<double>(diff) * scaled_two_pow_64 : 0.0;
670  }
671  }
672  }
673  else
674  {
675  double scaled_two_pow_64 = inv_scale;
676  for (std::size_t j = 0; j < coeff_mod_count;
677  j++, scaled_two_pow_64 *= two_pow_64)
678  {
679  auto curr_coeff = wide_tmp_dest[i * coeff_mod_count + j];
680  res[i] += curr_coeff ?
681  static_cast<double>(curr_coeff) * scaled_two_pow_64 : 0.0;
682  }
683  }
684 
685  // Scaling instead incorporated above; this can help in cases
686  // where otherwise pow(two_pow_64, j) would overflow due to very
687  // large coeff_mod_count and very large scale
688  // res[i] = res_accum * inv_scale;
689  }
690 
691  std::size_t tt = coeff_count;
692  for (int i = 0; i < logn; i++)
693  {
694  std::size_t mm = std::size_t(1) << i;
695  tt >>= 1;
696 
697  for (std::size_t j = 0; j < mm; j++)
698  {
699  std::size_t j1 = 2 * j * tt;
700  std::size_t j2 = j1 + tt - 1;
701  auto s = roots_[mm + j];
702 
703  for (std::size_t k = j1; k < j2 + 1; k++)
704  {
705  auto u = res[k];
706  auto v = res[k + tt] * s;
707  res[k] = u + v;
708  res[k + tt] = u - v;
709  }
710  }
711  }
712 
713  destination.clear();
714  destination.reserve(slots_);
715  for (std::size_t i = 0; i < slots_; i++)
716  {
717  destination.emplace_back(
718  from_complex<T>(res[matrix_reps_index_map_[i]]));
719  }
720  }

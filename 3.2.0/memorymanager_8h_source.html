514  template<typename... Args>
515  static inline MemoryPoolHandle GetPool(mm_prof_opt_t prof_opt, Args &&...args)
516  {
517  switch (prof_opt)
518  {
519  case mm_prof_opt::FORCE_GLOBAL:
520  return MemoryPoolHandle::Global();
521 
522  case mm_prof_opt::FORCE_NEW:
523  return MemoryPoolHandle::New(std::forward<Args>(args)...);
524 #ifndef _M_CEE
525  case mm_prof_opt::FORCE_THREAD_LOCAL:
526  return MemoryPoolHandle::ThreadLocal();
527 #endif
528  default:
529 #ifdef SEAL_DEBUG
530  {
531  auto pool = mm_prof_->get_pool(prof_opt);
532  if (!pool)
533  {
534  throw std::logic_error("cannot return uninitialized pool");
535  }
536  return pool;
537  }
538 #endif
539  return mm_prof_->get_pool(prof_opt);

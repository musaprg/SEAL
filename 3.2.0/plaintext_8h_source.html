410  inline bool operator ==(const Plaintext &compare) const
411  {
412  std::size_t sig_coeff_count = significant_coeff_count();
413  std::size_t sig_coeff_count_compare = compare.significant_coeff_count();
414  bool parms_id_compare = (is_ntt_form() && compare.is_ntt_form()
415  && (parms_id_ == compare.parms_id_)) ||
416  (!is_ntt_form() && !compare.is_ntt_form());
417  return parms_id_compare
418  && (sig_coeff_count == sig_coeff_count_compare)
419  && std::equal(data_.cbegin(),
420  data_.cbegin() + sig_coeff_count,
421  compare.data_.cbegin(),
422  compare.data_.cbegin() + sig_coeff_count)
423  && std::all_of(data_.cbegin() + sig_coeff_count,
424  data_.cend(), util::is_zero<pt_coeff_type>)
425  && std::all_of(compare.data_.cbegin() + sig_coeff_count,
426  compare.data_.cend(), util::is_zero<pt_coeff_type>)

424  inline bool operator ==(const Plaintext &compare) const
425  {
426  std::size_t sig_coeff_count = significant_coeff_count();
427  std::size_t sig_coeff_count_compare = compare.significant_coeff_count();
428  bool parms_id_compare = (is_ntt_form() && compare.is_ntt_form()
429  && (parms_id_ == compare.parms_id_)) ||
430  (!is_ntt_form() && !compare.is_ntt_form());
431  return parms_id_compare
432  && (sig_coeff_count == sig_coeff_count_compare)
433  && std::equal(data_.cbegin(),
434  data_.cbegin() + sig_coeff_count,
435  compare.data_.cbegin(),
436  compare.data_.cbegin() + sig_coeff_count)
437  && std::all_of(data_.cbegin() + sig_coeff_count,
438  data_.cend(), util::is_zero<pt_coeff_type>)
439  && std::all_of(compare.data_.cbegin() + sig_coeff_count,
440  compare.data_.cend(), util::is_zero<pt_coeff_type>)
441  && util::are_close(scale_, compare.scale_);

498  template<typename... Args>
499  static inline MemoryPoolHandle GetPool(mm_prof_opt_t prof_opt, Args &&...args)
500  {
501  switch (prof_opt)
502  {

504  return MemoryPoolHandle::Global();
505 
506  case mm_prof_opt::FORCE_NEW:

510  return MemoryPoolHandle::ThreadLocal();

512  default:

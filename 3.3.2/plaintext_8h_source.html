427  SEAL_NODISCARD inline bool operator ==(const Plaintext &compare) const
428  {
429  std::size_t sig_coeff_count = significant_coeff_count();
430  std::size_t sig_coeff_count_compare = compare.significant_coeff_count();
431  bool parms_id_compare = (is_ntt_form() && compare.is_ntt_form()
432  && (parms_id_ == compare.parms_id_)) ||
433  (!is_ntt_form() && !compare.is_ntt_form());
434  return parms_id_compare
435  && (sig_coeff_count == sig_coeff_count_compare)
436  && std::equal(data_.cbegin(),
437  data_.cbegin() + sig_coeff_count,
438  compare.data_.cbegin(),
439  compare.data_.cbegin() + sig_coeff_count)
440  && std::all_of(data_.cbegin() + sig_coeff_count,
441  data_.cend(), util::is_zero<pt_coeff_type>)
442  && std::all_of(compare.data_.cbegin() + sig_coeff_count,
443  compare.data_.cend(), util::is_zero<pt_coeff_type>)
444  && util::are_close(scale_, compare.scale_);
445  }
